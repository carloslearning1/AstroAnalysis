from flask import Flask,render_template,request
from astro import Astro
from genai import AiAnswer

app= Flask(__name__)

@app.route("/" )
def main_page():
    return render_template("index.html")

@app.route("/analysis" , methods=["POST"])
def analysis():
    if request.method == "POST":
        data = request.form.to_dict()
       
        int_keys = ["year","month", "date", "hours", "minutes", "seconds"]
        float_keys = ["latitude", "longitude", "timezone"]
        for k in int_keys:
            data[k] = int(data[k])
        for k in float_keys:
            data[k] = float(data[k])
        caller = Astro(data)
        response = caller.make_call()
        ai_caller = AiAnswer(response)
        ai_response = ai_caller.analyze()
        


        return render_template("analysis.html",ai_response=ai_response)
    else:
        return render_template("index.html")



if __name__ == "__main__":
    app.run(debug=True)
